function solution(left, right) { let l = left; let r = right; let answer = 0; let plus = 0; let minus = 0; for (; l <= r; l++) { let leftcount = 0; for (let i = 0; i <= l; i++) { if (l % i == 0) { leftcount += 1; } } if (leftcount % 2 == 0) { plus += l; } else { minus += l; } } answer = (plus - minus); return answer; }
